  <title>Redirecting to Facebook Post...</title>
  <script>
    function openFacebookPost() {
      const userAgent = navigator.userAgent || navigator.vendor || window.opera;

      if (/android/i.test(userAgent)) {
        // Android → ใช้ Intent เปิดโพสต์
        window.location.href = "intent://post/123456789012345#Intent;scheme=fb;package=com.facebook.katana;end;";
      } else if (/iPhone|iPad|iPod/.test(userAgent)) {
        // iOS → เปิดผ่าน fb:// แล้ว fallback
        window.location.href = "fb://post/123456789012345";
        setTimeout(function () {
          window.location.href = "https://www.facebook.com/Sahaphanautosale/posts/123456789012345";
        }, 2000);
      } else {
        // คอมพิวเตอร์ → เปิดผ่านเว็บ
        window.location.href = "https://www.facebook.com/Sahaphanautosale/posts/123456789012345";
      }
    }
  </script>
</head>
<body onload="openFacebookPost()">
